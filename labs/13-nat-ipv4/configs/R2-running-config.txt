! R2 Running Configuration - NAT for IPv4
! George Brown College - CCNA Lab 13
!
hostname R2
!
interface Serial0/1/0
 description WAN to ISP (Outside NAT)
 ip address 209.165.200.225 255.255.255.224
 ip nat outside
 no shutdown
!
interface GigabitEthernet0/0/0
 description LAN Subnet A (Inside NAT)
 ip address 192.168.1.1 255.255.255.0
 ip nat inside
 no shutdown
!
interface GigabitEthernet0/0/1
 description LAN Subnet B (Inside NAT)
 ip address 192.168.2.1 255.255.255.0
 ip nat inside
 no shutdown
!
! Dynamic NAT pool
ip nat pool NAT-POOL1 209.165.200.226 209.165.200.228 netmask 255.255.255.224
!
! ACL to define inside local addresses
ip access-list standard NAT-ACL
 permit 192.168.1.0 0.0.0.255
 permit 192.168.2.0 0.0.0.255
!
! NAT overload (PAT) on pool
ip nat inside source list NAT-ACL pool NAT-POOL1
!
! Static NAT entry for DMZ server
ip nat inside source static 192.168.2.20 209.165.200.228
!
! Default route to ISP
ip route 0.0.0.0 0.0.0.0 209.165.200.254
!
end
