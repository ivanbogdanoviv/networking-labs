! R1 Running Configuration - Static Routes IPv4 & IPv6
! George Brown College - CCNA Lab 09
!
hostname R1
!
ipv6 unicast-routing
!
interface GigabitEthernet0/0/0
 description LAN - PC-A network
 ip address 192.168.1.1 255.255.255.0
 ipv6 address 2001:DB8:1::1/64
 ipv6 address FE80::1 link-local
 no shutdown
!
interface GigabitEthernet0/0/1
 description WAN link to R2
 ip address 10.1.1.1 255.255.255.252
 ipv6 address 2001:DB8:A::1/64
 ipv6 address FE80::1 link-local
 no shutdown
!
! IPv4 Static Routes
ip route 192.168.2.0 255.255.255.0 10.1.1.2
ip route 192.168.3.0 255.255.255.0 10.1.1.2
!
! IPv6 Static Routes
ipv6 route 2001:DB8:2::/64 2001:DB8:A::2
ipv6 route 2001:DB8:3::/64 2001:DB8:A::2
!
end
